    CREATE OR REPLACE FUNCTION public.TestProcedure4 (p_tab public.classifier_rec [], refcursor)
RETURNS refcursor AS
$$
begin
    open $2 for select Id, Name from unnest(p_tab);
    return $2;
end
$$
